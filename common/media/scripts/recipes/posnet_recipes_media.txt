module PhobosOperationalSignals {

/* ----- REFURBISH VHS TAPE ----- */
/*  Repair a damaged VHS tape using duct tape and a screwdriver.             */

    craftRecipe RefurbishVHSTape
    {
        time = 200,
        category = Phobos Field Ops,
        NeedToBeLearn = false,
        OnCreate = POS_CraftCallbacks.onCreateRefurbishedTape,

        Tags = AnySurfaceCraft,
        timedAction = Making_Surface,

        inputs
        {
            item 1 [PhobosOperationalSignals.DamagedVHSTape],
            item 1 [Base.DuctTape],
            item 1 [Base.Screwdriver] mode:keep flags[NoBrokenItems],
        }
        outputs
        {
            item 1 PhobosOperationalSignals.RefurbishedVHSCTape,
        }
    }

/* ----- SPLICE RECON TAPE ----- */
/*  Combine two damaged VHS tapes into a single usable spliced tape.         */

    craftRecipe SpliceReconTape
    {
        time = 250,
        category = Phobos Field Ops,
        NeedToBeLearn = false,
        OnCreate = POS_CraftCallbacks.onCreateSplicedTape,

        Tags = AnySurfaceCraft,
        timedAction = Making_Surface,

        inputs
        {
            item 2 [PhobosOperationalSignals.DamagedVHSTape],
            item 1 [Base.DuctTape],
        }
        outputs
        {
            item 1 PhobosOperationalSignals.SplicedReconTape,
        }
    }

    }

    {
        time = 180,
        category = Phobos SIGINT Ops,

        Tags = POS_TVStation;CannotBeResearched,
        timedAction = Craft,

        inputs

            item 1 [Base.VHS_Retail;Base.VHS_Home],
            item 1 [Base.Screwdriver] mode:keep flags[NoBrokenItems],
        }
        outputs
        {
            item 1 PhobosOperationalSignals.BlankVHSCTape,
        }
    }

    {
